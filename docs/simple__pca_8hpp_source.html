81template<bool sparse_, typename Value_, typename Index_, class EigenVector_>
82void compute_row_means_and_variances(const tatami::Matrix<Value_, Index_>* mat, int num_threads, EigenVector_& center_v, EigenVector_& scale_v) {
83 if (mat->prefer_rows()) {
84 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
85 tatami::Options opt;
86 opt.sparse_extract_index = false;
87 auto ext = tatami::consecutive_extractor<sparse_>(mat, true, start, length, opt);
88 auto ncells = mat->ncol();
89 std::vector<Value_> vbuffer(ncells);
90
91 for (Index_ r = start, end = start + length; r < end; ++r) {
92 auto results = [&]() {
93 if constexpr(sparse_) {
94 auto range = ext->fetch(vbuffer.data(), NULL);
95 return tatami_stats::variances::direct(range.value, range.number, ncells, /* skip_nan = */ false);
96 } else {
97 auto ptr = ext->fetch(vbuffer.data());
98 return tatami_stats::variances::direct(ptr, ncells, /* skip_nan = */ false);
99 }
100 }();
101 center_v.coeffRef(r) = results.first;
102 scale_v.coeffRef(r) = results.second;
103 }
104 }, mat->nrow(), num_threads);
105
106 } else {
107 tatami::parallelize([&](size_t t, Index_ start, Index_ length) -> void {
108 tatami::Options opt;
109 auto ncells = mat->ncol();
110 auto ext = tatami::consecutive_extractor<sparse_>(mat, false, 0, ncells, start, length, opt);
111
112 typedef typename EigenVector_::Scalar Scalar;
113 tatami_stats::LocalOutputBuffer<Scalar> cbuffer(t, start, length, center_v.data());
114 tatami_stats::LocalOutputBuffer<Scalar> sbuffer(t, start, length, scale_v.data());
115
116 auto running = [&]() {
117 if constexpr(sparse_) {
118 return tatami_stats::variances::RunningSparse<Scalar, Value_, Index_>(length, cbuffer.data(), sbuffer.data(), /* skip_nan = */ false, /* subtract = */ start);
119 } else {
120 return tatami_stats::variances::RunningDense<Scalar, Value_, Index_>(length, cbuffer.data(), sbuffer.data(), /* skip_nan = */ false);
121 }
122 }();
123
124 std::vector<Value_> vbuffer(length);
125 typename std::conditional<sparse_, std::vector<Index_>, Index_>::type ibuffer(length);
126 for (Index_ r = 0; r < ncells; ++r) {
127 if constexpr(sparse_) {
128 auto range = ext->fetch(vbuffer.data(), ibuffer.data());
129 running.add(range.value, range.index, range.number);
130 } else {
131 auto ptr = ext->fetch(vbuffer.data());
132 running.add(ptr);
133 }
134 }
135
136 running.finish();
137 cbuffer.transfer();
138 sbuffer.transfer();
139 }, mat->nrow(), num_threads);
140 }

163template<typename Value_, typename Index_, class EigenMatrix_, class EigenVector_>
164void run_sparse(
165 const tatami::Matrix<Value_, Index_>* mat,
166 int rank,
167 const Options& options,
168 EigenMatrix_& components,
169 EigenMatrix_& rotation,
170 EigenVector_& variance_explained,
171 EigenVector_& center_v,
172 EigenVector_& scale_v,
173 typename EigenVector_::Scalar& total_var)
174{
175 Index_ ngenes = mat->nrow();
176 center_v.resize(ngenes);
177 scale_v.resize(ngenes);
178
179 if (options.realize_matrix) {
180 // 'extracted' contains row-major contents...
181 auto extracted = tatami::retrieve_compressed_sparse_contents<Value_, Index_>(
182 mat,
183 /* row = */ true,
184 /* two_pass = */ false,
185 /* threads = */ options.num_threads
186 );
187
188 // But we effectively transpose it to CSC with genes in columns.
189 Index_ ncells = mat->ncol();
190 irlba::ParallelSparseMatrix emat(
191 ncells,
192 ngenes,
193 std::move(extracted.value),
194 std::move(extracted.index),
195 std::move(extracted.pointers),
196 true,
197 options.num_threads
198 );
199
200 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
201 const auto& ptrs = emat.get_pointers();
202 const auto& values = emat.get_values();
203 for (size_t r = start, end = start + length; r < end; ++r) {
204 auto offset = ptrs[r];
205 Index_ num_nonzero = ptrs[r + 1] - offset;
206 auto results = tatami_stats::variances::direct(values.data() + offset, num_nonzero, ncells, /* skip_nan = */ false);
207 center_v.coeffRef(r) = results.first;
208 scale_v.coeffRef(r) = results.second;
209 }
210 }, ngenes, options.num_threads);
211
212 total_var = pca_utils::process_scale_vector(options.scale, scale_v);
213 run_irlba_deferred(emat, rank, options, components, rotation, variance_explained, center_v, scale_v);
214
215 } else {
216 compute_row_means_and_variances<true>(mat, options.num_threads, center_v, scale_v);
217 total_var = pca_utils::process_scale_vector(options.scale, scale_v);
218 run_irlba_deferred(
219 pca_utils::TransposedTatamiWrapper<EigenVector_, Value_, Index_>(mat, options.num_threads),
220 rank,
221 options,
222 components,
223 rotation,
224 variance_explained,
225 center_v,

231template<typename Value_, typename Index_, class EigenMatrix_, class EigenVector_>
232void run_dense(
233 const tatami::Matrix<Value_, Index_>* mat,
234 int rank,
235 const Options& options,
236 EigenMatrix_& components,
237 EigenMatrix_& rotation,
238 EigenVector_& variance_explained,
239 EigenVector_& center_v,
240 EigenVector_& scale_v,
241 typename EigenVector_::Scalar& total_var)
242{
243 Index_ ngenes = mat->nrow();
244 center_v.resize(ngenes);
245 scale_v.resize(ngenes);
246
247 if (options.realize_matrix) {
248 // Create a matrix with genes in columns.
249 Index_ ncells = mat->ncol();
250 EigenMatrix_ emat(ncells, ngenes);
251
252 // If emat is row-major, we want to fill it with columns of 'mat', so row_major = false.
253 // If emat is column-major, we want to fill it with rows of 'mat', so row_major = true.
254 tatami::convert_to_dense(mat, /* row_major = */ !emat.IsRowMajor, emat.data(), options.num_threads);
255
256 center_v.array() = emat.array().colwise().sum();
257 if (ncells) {
258 center_v /= ncells;
259 } else {
260 std::fill(center_v.begin(), center_v.end(), std::numeric_limits<typename EigenVector_::Scalar>::quiet_NaN());
261 }
262 emat.array().rowwise() -= center_v.adjoint().array(); // applying it to avoid wasting time with deferred operations inside IRLBA.
263
264 scale_v.array() = emat.array().colwise().squaredNorm();
265 if (ncells > 1) {
266 scale_v /= ncells - 1;
267 } else {
268 std::fill(scale_v.begin(), scale_v.end(), std::numeric_limits<typename EigenVector_::Scalar>::quiet_NaN());
269 }
270
271 total_var = pca_utils::process_scale_vector(options.scale, scale_v);
272 if (options.scale) {
273 emat.array().rowwise() /= scale_v.adjoint().array();
274 }
275
276 irlba::compute(emat, rank, components, rotation, variance_explained, options.irlba_options);
277
278 } else {
279 compute_row_means_and_variances<false>(mat, options.num_threads, center_v, scale_v);
280 total_var = pca_utils::process_scale_vector(options.scale, scale_v);
281 run_irlba_deferred(
282 pca_utils::TransposedTatamiWrapper<EigenVector_, Value_, Index_>(mat, options.num_threads),
283 rank,
284 options,
285 components,
286 rotation,
287 variance_explained,
288 center_v,
289 scale_v
290 );
291 }

294template<typename Value_, typename Index_, class EigenMatrix_, class EigenVector_>
295void dispatch(
296 const tatami::Matrix<Value_, Index_>* mat,
297 int rank,
298 const Options& options,
299 EigenMatrix_& components,
300 EigenMatrix_& rotation,
301 EigenVector_& variance_explained,
302 EigenVector_& center_v,
303 EigenVector_& scale_v,

307

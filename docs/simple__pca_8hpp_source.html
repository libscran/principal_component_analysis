77template<bool sparse_, typename Value_, typename Index_, class EigenVector_>
78void compute_row_means_and_variances(const tatami::Matrix<Value_, Index_>& mat, int num_threads, EigenVector_& center_v, EigenVector_& scale_v) {
79 if (mat.prefer_rows()) {
80 tatami::parallelize([&](size_t, Index_ start, Index_ length) -> void {
81 tatami::Options opt;
82 opt.sparse_extract_index = false;
83 auto ext = tatami::consecutive_extractor<sparse_>(&mat, true, start, length, opt);
84 auto ncells = mat.ncol();
85 std::vector<Value_> vbuffer(ncells);
86
87 for (Index_ r = start, end = start + length; r < end; ++r) {
88 auto results = [&]() {
89 if constexpr(sparse_) {
90 auto range = ext->fetch(vbuffer.data(), NULL);
91 return tatami_stats::variances::direct(range.value, range.number, ncells, /* skip_nan = */ false);
92 } else {
93 auto ptr = ext->fetch(vbuffer.data());
94 return tatami_stats::variances::direct(ptr, ncells, /* skip_nan = */ false);
95 }
96 }();
97 center_v.coeffRef(r) = results.first;
98 scale_v.coeffRef(r) = results.second;
99 }
100 }, mat.nrow(), num_threads);
101
102 } else {
103 tatami::parallelize([&](size_t t, Index_ start, Index_ length) -> void {
104 tatami::Options opt;
105 auto ncells = mat.ncol();
106 auto ext = tatami::consecutive_extractor<sparse_>(&mat, false, 0, ncells, start, length, opt);
107
108 typedef typename EigenVector_::Scalar Scalar;
109 tatami_stats::LocalOutputBuffer<Scalar> cbuffer(t, start, length, center_v.data());
110 tatami_stats::LocalOutputBuffer<Scalar> sbuffer(t, start, length, scale_v.data());
111
112 auto running = [&]() {
113 if constexpr(sparse_) {
114 return tatami_stats::variances::RunningSparse<Scalar, Value_, Index_>(length, cbuffer.data(), sbuffer.data(), /* skip_nan = */ false, /* subtract = */ start);
115 } else {
116 return tatami_stats::variances::RunningDense<Scalar, Value_, Index_>(length, cbuffer.data(), sbuffer.data(), /* skip_nan = */ false);
117 }
118 }();
119
120 std::vector<Value_> vbuffer(length);
121 typename std::conditional<sparse_, std::vector<Index_>, Index_>::type ibuffer(length);
122 for (Index_ r = 0; r < ncells; ++r) {
123 if constexpr(sparse_) {
124 auto range = ext->fetch(vbuffer.data(), ibuffer.data());
125 running.add(range.value, range.index, range.number);
126 } else {
127 auto ptr = ext->fetch(vbuffer.data());
128 running.add(ptr);
129 }
130 }
131
132 running.finish();
133 cbuffer.transfer();
134 sbuffer.transfer();
135 }, mat.nrow(), num_threads);
136 }
137}

158template<typename Value_, typename Index_, class EigenMatrix_, class EigenVector_>
159void run_sparse(
160 const tatami::Matrix<Value_, Index_>& mat,
161 const SimplePcaOptions& options,
162 EigenMatrix_& components,
163 EigenMatrix_& rotation,
164 EigenVector_& variance_explained,
165 EigenVector_& center_v,
166 EigenVector_& scale_v,
167 typename EigenVector_::Scalar& total_var)
168{
169 Index_ ngenes = mat.nrow();
170 center_v.resize(ngenes);
171 scale_v.resize(ngenes);
172
173 if (options.realize_matrix) {
174 // 'extracted' contains row-major contents...
175 auto extracted = tatami::retrieve_compressed_sparse_contents<Value_, Index_>(
176 &mat,
177 /* row = */ true,
178 /* two_pass = */ false,
179 /* threads = */ options.num_threads
180 );
181
182 // But we effectively transpose it to CSC with genes in columns.
183 Index_ ncells = mat.ncol();
184 irlba::ParallelSparseMatrix emat(
185 ncells,
186 ngenes,
187 std::move(extracted.value),
188 std::move(extracted.index),
189 std::move(extracted.pointers),
190 true,
191 options.num_threads
192 );
193
194 tatami::parallelize([&](size_t, size_t start, size_t length) -> void {
195 const auto& ptrs = emat.get_pointers();
196 const auto& values = emat.get_values();
197 for (size_t r = start, end = start + length; r < end; ++r) {
198 auto offset = ptrs[r];
199 Index_ num_nonzero = ptrs[r + 1] - offset;
200 auto results = tatami_stats::variances::direct(values.data() + offset, num_nonzero, ncells, /* skip_nan = */ false);
201 center_v.coeffRef(r) = results.first;
202 scale_v.coeffRef(r) = results.second;
203 }
204 }, ngenes, options.num_threads);
205
206 total_var = internal::process_scale_vector(options.scale, scale_v);
207 run_irlba_deferred(emat, options, components, rotation, variance_explained, center_v, scale_v);
208
209 } else {
210 compute_row_means_and_variances<true>(mat, options.num_threads, center_v, scale_v);
211 total_var = internal::process_scale_vector(options.scale, scale_v);
212 run_irlba_deferred(
213 internal::TransposedTatamiWrapper<EigenVector_, Value_, Index_>(mat, options.num_threads),
214 options,
215 components,
216 rotation,
217 variance_explained,
218 center_v,
219 scale_v
220 );
221 }

224template<typename Value_, typename Index_, class EigenMatrix_, class EigenVector_>
225void run_dense(
226 const tatami::Matrix<Value_, Index_>& mat,
227 const SimplePcaOptions& options,
228 EigenMatrix_& components,
229 EigenMatrix_& rotation,
230 EigenVector_& variance_explained,
231 EigenVector_& center_v,
232 EigenVector_& scale_v,
233 typename EigenVector_::Scalar& total_var)
234{
235 Index_ ngenes = mat.nrow();
236 center_v.resize(ngenes);
237 scale_v.resize(ngenes);
238
239 if (options.realize_matrix) {
240 // Create a matrix with genes in columns.
241 Index_ ncells = mat.ncol();
242 EigenMatrix_ emat(ncells, ngenes);
243
244 // If emat is row-major, we want to fill it with columns of 'mat', so row_major = false.
245 // If emat is column-major, we want to fill it with rows of 'mat', so row_major = true.
246 tatami::convert_to_dense(&mat, /* row_major = */ !emat.IsRowMajor, emat.data(), options.num_threads);
247
248 center_v.array() = emat.array().colwise().sum();
249 if (ncells) {
250 center_v /= ncells;
251 } else {
252 std::fill(center_v.begin(), center_v.end(), std::numeric_limits<typename EigenVector_::Scalar>::quiet_NaN());
253 }
254 emat.array().rowwise() -= center_v.adjoint().array(); // applying it to avoid wasting time with deferred operations inside IRLBA.
255
256 scale_v.array() = emat.array().colwise().squaredNorm();
257 if (ncells > 1) {
258 scale_v /= ncells - 1;
259 } else {
260 std::fill(scale_v.begin(), scale_v.end(), std::numeric_limits<typename EigenVector_::Scalar>::quiet_NaN());
261 }
262
263 total_var = internal::process_scale_vector(options.scale, scale_v);
264 if (options.scale) {
265 emat.array().rowwise() /= scale_v.adjoint().array();
266 }
267
268 irlba::compute(emat, options.number, components, rotation, variance_explained, options.irlba_options);
269
270 } else {
271 compute_row_means_and_variances<false>(mat, options.num_threads, center_v, scale_v);
272 total_var = internal::process_scale_vector(options.scale, scale_v);
273 run_irlba_deferred(
274 internal::TransposedTatamiWrapper<EigenVector_, Value_, Index_>(mat, options.num_threads),
275 options,
276 components,
277 rotation,
278 variance_explained,
279 center_v,
280 scale_v
281 );
282 }
283}

356 irlba::EigenThreadScope t(options.num_threads);
357
358 if (mat.sparse()) {

360 } else {

370 output.scale = EigenVector_();
371 }
